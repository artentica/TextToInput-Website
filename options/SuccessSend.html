$('#basicTableModified').TextToInput({
    mainButtonDiv : "#basicTableModified div:nth-child(1)",
    link : "#",
    SuccessSend: function (json) { //I overload the success function so that the outcome is displayed in a green
      $("#basicTableModified .result").empty();
      $("#basicTableModified .result").append(json);
      $("#basicTableModified").data('TextToInput').resetChange();
      $("#basicTableModified").attr("style","background-color:#8FBC8F");
    }
});
